import pandas as pd

data = pd.read_csv(url)

# Selected variables
selected_variables = [
    "povarea",  # Neighborhood poverty
    "agegp",    # Age at HCT (Categorical)
    "age",      # Age at HCT (Continuous)
    "racegp",   # Recipient race
    "agvhd",    # Acute GVHD
    "condintb", # Conditioning intensity
    "drigp",    # Refined disease risk index
    "hctcigpc", # HCT-CI
